main.tf


    
 provider "aws" {
  region = "us-east-1"  # Change this to your preferred region
  access_key = "xyz"         # üîÅ Replace with your actual access key
  secret_key = "xyz"         # üîÅ Replace with your actual secret key
}

# Create VPC
resource "aws_vpc" "main" {
  cidr_block = "10.0.0.0/16"
}

# Create Subnet
resource "aws_subnet" "main" {
  vpc_id     = aws_vpc.main.id
  cidr_block = "10.0.1.0/24"
}

# Create Security Group allowing SSH
resource "aws_security_group" "allow_ssh" {
  name        = "allow_ssh"
  description = "Allow SSH"
  vpc_id      = aws_vpc.main.id

  ingress {
    from_port   = 22
    to_port     = 22
    protocol    = "tcp"
    cidr_blocks = ["0.0.0.0/0"]
  }

  egress {
    from_port   = 0
    to_port     = 0
    protocol    = "-1"
    cidr_blocks = ["0.0.0.0/0"]
  }
}

# Create EC2 Instance and attach user data to mount EBS volume
resource "aws_instance" "web" {
  ami                    = "ami-0150ccaf51ab55a51"  # Update with latest valid AMI in your region
  instance_type          = "t2.micro"
  subnet_id              = aws_subnet.main.id
  vpc_security_group_ids = [aws_security_group.allow_ssh.id]
  key_name               = var.key_name

  user_data = <<-EOF
              #!/bin/bash
              mkfs -t ext4 /dev/xvdf
              mkdir /data
              mount /dev/xvdf /data
              echo "/dev/xvdf /data ext4 defaults,nofail 0 2" >> /etc/fstab
            EOF

  tags = {
    Name = "EBSDemoInstance"
  }
}

# Create EBS Volume
resource "aws_ebs_volume" "ebs" {
  availability_zone = aws_instance.web.availability_zone
  size              = 8
  type              = "gp2"

  tags = {
    Name = "MyEBSVolume"
  }
}

# Attach EBS Volume to EC2
resource "aws_volume_attachment" "ebs_attach" {
  device_name = "/dev/xvdf"
  volume_id   = aws_ebs_volume.ebs.id
  instance_id = aws_instance.web.id
  force_detach = true
}
